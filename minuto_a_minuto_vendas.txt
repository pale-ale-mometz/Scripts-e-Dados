WITH RECURSIVE all_hours AS (
    SELECT 0 AS h
    UNION ALL SELECT h + 1 FROM all_hours WHERE h < 23
),
all_minutes AS (
    SELECT 0 AS m
    UNION ALL SELECT m + 1 FROM all_minutes WHERE m < 59
),
-- Full 24h × 60min grid (1440 rows)
full_grid AS (
    SELECT h.h AS sale_hour, m.m AS sale_minute
    FROM all_hours h
    CROSS JOIN all_minutes m
),
-- Actual data
actual AS (
    SELECT
        HOUR(dt_filiacao)   AS sale_hour,
        MINUTE(dt_filiacao) AS sale_minute,
        COUNT(DISTINCT CONCAT(cpf, '|', DATE(dt_filiacao))) AS total_sales
    FROM mysql_bi_g.NOMINAL_VENDAS_DIARIO nvd
    GROUP BY HOUR(dt_filiacao), MINUTE(dt_filiacao)
)
SELECT
    g.sale_hour,
    g.sale_minute,
    COALESCE(a.total_sales, 0)                          AS total_sales,
    CASE WHEN a.total_sales IS NULL THEN '⚠ missing' 
         WHEN a.total_sales < 2    THEN '↓ low'
         ELSE 'ok'
    END                                                  AS status
FROM full_grid g
LEFT JOIN actual a
    ON g.sale_hour   = a.sale_hour
    AND g.sale_minute = a.sale_minute
ORDER BY g.sale_hour, g.sale_minute;
